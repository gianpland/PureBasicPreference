Procedure AlertThread(Parameter)

  Repeat
    Debug "Alert !"
    Delay(3000)
  ForEver

EndProcedure

CreateThread(@AlertThread(), 154)

MessageRequester("Info", "It will display an alert every 3 seconds."+#LF$+"Click To finish the program", 0)
